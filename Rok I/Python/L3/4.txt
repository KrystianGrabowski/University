from math import sqrt
def pierwsza(liczba):
    pomoc=sqrt(liczba)
    b = 0
    l = 2
    while l <= pomoc:
        reszta = liczba % l
        if reszta == 0:
            b = b + 1
        l = l + 1
    if b == 0:
        return 1
    else:
        return 0

czyt = 1000000000
while czyt < 10000000000:
    if "7777777" in str(czyt):
        if pierwsza(czyt) == 1:
            print(czyt)
    czyt=czyt+1
